    <script>
        class BookingSystemUI {
            constructor() {
                this.sidebar = document.querySelector('.sidebar');
                this.menuToggle = document.querySelector('.menu-toggle');
                this.sidebarClose = document.querySelector('.sidebar-close');
                this.cancelModal = document.getElementById('cancelModal');
                this.currentBookingId = null;
                this.init();
            }

            init() {
                this.bindEvents();
                this.autoHideFlashMessages();
                this.setupEquipmentCellExpansion();
            }

            bindEvents() {
                // Toggle sidebar
                this.menuToggle?.addEventListener('click', () => this.toggleSidebar());
                this.sidebarClose?.addEventListener('click', () => this.closeSidebar());

                // Handle navigation
                document.querySelectorAll('.sidebar-menu a[data-target]').forEach(link => {
                    link.addEventListener('click', (e) => this.handleNavigation(e));
                });

                // Close sidebar on outside click (mobile)
                document.addEventListener('click', (e) => {
                    if (window.innerWidth < 992 &&
                        this.sidebar.classList.contains('active') &&
                        !this.sidebar.contains(e.target) &&
                        !this.menuToggle.contains(e.target)) {
                        this.closeSidebar();
                    }
                });

                // Handle escape key
                document.addEventListener('keydown', (e) => {
                    if (e.key === 'Escape' && this.sidebar.classList.contains('active')) {
                        this.closeSidebar();
                    }
                });

                // Cancel booking buttons
                document.querySelectorAll('.btn-cancel').forEach(button => {
                    button.addEventListener('click', (e) => this.showCancelModal(e));
                });

                // Modal buttons
                document.getElementById('confirmCancel')?.addEventListener('click', () => this.cancelBooking());
                document.getElementById('closeModal')?.addEventListener('click', () => this.hideModal());

                // Close modal when clicking outside
                this.cancelModal?.addEventListener('click', (e) => {
                    if (e.target === this.cancelModal) {
                        this.hideModal();
                    }
                });
            }
            
            setupEquipmentCellExpansion() {
                // Add click event to all expandable equipment cells
                document.querySelectorAll('.equipment-cell.expandable').forEach(cell => {
                    cell.addEventListener('click', (e) => {
                        e.stopPropagation();
                        cell.classList.toggle('expanded');
                    });
                });
                
                // Close expanded cells when clicking elsewhere
                document.addEventListener('click', (e) => {
                    if (!e.target.closest('.equipment-cell')) {
                        document.querySelectorAll('.equipment-cell.expanded').forEach(cell => {
                            cell.classList.remove('expanded');
                        });
                    }
                });
            }

            toggleSidebar() {
                this.sidebar.classList.toggle('active');
            }

            closeSidebar() {
                this.sidebar.classList.remove('active');
            }

            handleNavigation(e) {
                e.preventDefault();
                const link = e.currentTarget;
                const target = link.getAttribute('data-target');

                if (!target) return;

                // Update active states
                this.updateActiveStates(link);

                // Show target content
                this.showContent(target);

                // Update page title
                this.updatePageTitle(link.textContent.trim());

                // Close sidebar on mobile
                if (window.innerWidth < 992) {
                    this.closeSidebar();
                }
            }

            updateActiveStates(activeLink) {
                document.querySelectorAll('.sidebar-menu a').forEach(link => {
                    link.classList.remove('active');
                });
                activeLink.classList.add('active');
            }

            showContent(targetId) {
                document.querySelectorAll('.dashboard-content').forEach(section => {
                    section.classList.remove('active');
                });

                const targetSection = document.getElementById(targetId);
                if (targetSection) {
                    targetSection.classList.add('active');
                }
            }

            updatePageTitle(title) {
                const pageTitle = document.querySelector('.page-title');
                if (pageTitle) {
                    pageTitle.textContent = title;
                }
            }

            showCancelModal(e) {
                e.preventDefault();
                const button = e.currentTarget;
                this.currentBookingId = button.getAttribute('data-booking-id');
                this.cancelModal.style.display = 'flex';
            }

            hideModal() {
                this.cancelModal.style.display = 'none';
                this.currentBookingId = null;
            }

            async cancelBooking() {
                if (!this.currentBookingId) return;

                try {
                    // Show loading state
                    const confirmButton = document.getElementById('confirmCancel');
                    const originalText = confirmButton.textContent;
                    confirmButton.innerHTML = '<span class="loading"></span> Cancelling...';
                    confirmButton.disabled = true;

                    // Send cancel request to server
                    const response = await fetch('/cancel_booking', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({
                            booking_id: this.currentBookingId
                        })
                    });

                    const result = await response.json();

                    if (result.success) {
                        // Show success message and reload the page
                        this.showFlashMessage('Booking cancelled successfully!', 'success');
                        setTimeout(() => {
                            location.reload();
                        }, 1500);
                    } else {
                        this.showFlashMessage(result.message || 'Error cancelling booking', 'error');
                    }
                } catch (error) {
                    this.showFlashMessage('Network error. Please try again.', 'error');
                } finally {
                    this.hideModal();
                }
            }

            showFlashMessage(message, type) {
                const flashContainer = document.getElementById('flash-messages') || document.createElement('div');
                flashContainer.id = 'flash-messages';

                const flashMessage = document.createElement('div');
                flashMessage.className = `flash ${type}`;
                flashMessage.textContent = message;

                flashContainer.appendChild(flashMessage);
                document.body.appendChild(flashContainer);

                this.autoHideFlashMessages();
            }

            autoHideFlashMessages() {
                const flashContainer = document.getElementById('flash-messages');
                if (flashContainer) {
                    setTimeout(() => {
                        flashContainer.style.animation = 'fadeOut 0.5s forwards';
                        setTimeout(() => {
                            if (flashContainer.parentNode) {
                                flashContainer.parentNode.removeChild(flashContainer);
                            }
                        }, 500);
                    }, 3000);
                }
            }
        }

        // Initialize the application
        document.addEventListener('DOMContentLoaded', () => {
            new BookingSystemUI();
        });

        // Add smooth scrolling for better UX
        document.documentElement.style.scrollBehavior = 'smooth';

        // Optimize images loading
        document.querySelectorAll('img').forEach(img => {
            img.loading = 'lazy';
        });
    </script>
